#Экономное вычисление выражений в польской записи
Требуется составить программу, вычисляющую количество операций, которые нужно выполнить для экономного вычисления выражения.

x

	0

($xy)

	1

($(@ab)c)

	2

(#i($jk))

	2

(#($ab)($ab))

	2

(@(#ab)($ab))

	3

(#($a($b($cd)))(@($b($cd))($a($b($cd)))))

	5

(#($(#xy)($(#ab)(#ab)))(@z($(#ab)(#ab))))

	6
